function getArabicDayName(dateValue) {
    if (!dateValue) return "";

    const parts = dateValue.split("-").map(Number);
    if (parts.length !== 3) return "";

    const date = new Date(year, month - 1, day);

    const days = [

        "الثلاثاء",
        "الأربعاء",

function getTripDayIndex(dateValue) {

function formatTime(timeValue) {
    if (!timeValue) return "غير محدد";

    const parts = String(timeValue).split(":");
    if (parts.length < 2) return timeValue;

    const hour = Number(parts[0]);
    const minute = Number(parts[1]);

    if (isNaN(hour) || isNaN(minute)) return timeValue;

    const period = hour >= 12 ? "م" : "ص";
    const hour12 = hour % 12 || 12;

    return `${hour12}:${pad(minute)} ${period}`;
}

function formatDateTime(value) {
    if (!value) return "غير محدد";

    const date = new Date(value);
    if (isNaN(date.getTime())) return value;

    return date.toLocaleString("ar-EG", {
        weekday: "long",
        year: "numeric",
        month: "2-digit",
        day: "2-digit",
        hour: "2-digit",
        minute: "2-digit"
    });
}

    const assign = getAssignById(trip?.assign_id);

    if (assign) {
        const stations = getAssignStationsArray(assign);
        const routeId = getAssignRouteId(assign);
        const routeName = getAssignRouteName(assign);
        const finalStationId = getFinalDropoffStationId(assign);

        return {
            assign,
            assign_id: Number(assign.id),
            route_id: routeId,
            route_name: routeName,
            station_ids: stations,
            route_final_station_id: finalStationId,
            route_final_station_name: finalStationId !== null ? getStationName(finalStationId) : "غير محددة"
        };
    }

    const fallbackRouteId = trip?.route_id ? Number(trip.route_id) : null;
    const fallbackRoute = fallbackRouteId ? getRouteById(fallbackRouteId) : null;

    return {
        assign: null,
        assign_id: trip?.assign_id ? Number(trip.assign_id) : null,
        route_id: fallbackRouteId,
        route_name: fallbackRoute ? fallbackRoute.route_name : (fallbackRouteId ? `مسار رقم ${fallbackRouteId}` : "غير محدد"),
        station_ids: [],
        route_final_station_id: null,
        route_final_station_name: "غير محددة"
    };
}

function getStudentAbsenceInfo(studentId, tripDate) {
    const requests = getStudentAbsenceRequestsForDate(studentId, tripDate);

    const types = new Set(
        requests
            .map(request => normalizeText(request.type))
            .filter(Boolean)
    );

    const hasPickup = types.has("pickup");
    const hasDropoff = types.has("dropoff");
    const hasFull = types.has("full") || (hasPickup && hasDropoff);
    const hasAbsence = hasFull || hasPickup || hasDropoff;

    let label = "لا يوجد";
    if (hasFull) label = "طلب غياب كامل";
    else if (hasPickup) label = "طلب غياب صعود";
    else if (hasDropoff) label = "طلب غياب نزول";

    return {
        hasAbsence,
        hasFull,
        hasPickup,
        hasDropoff,
        affectsPickup: hasFull || hasPickup,
        affectsDropoff: hasFull || hasDropoff,
        label,
        requests
    };
}

    }

    if (absenceFilter === "exclude_dropoff") {
        return !absenceInfo.affectsDropoff;
    }

    return true;
}

function getStudentsForRoute() {
    const assignId = getEl("trip-route-select")?.value || "";
    const routeFilterValue = getTripStudentFilterValue();
    const dayFilterMode = getEl("trip-student-days-filter")?.value || "all_days";
    const genderFilterValue = getEl("trip-student-gender-filter")?.value || "all";
    const tripDate = getEl("trip-date-input")?.value || "";

    const selectedAssign = getAssignById(assignId);
    if (!selectedAssign) return [];

    const stationIds = getAssignStationsArray(selectedAssign);
    const finalDropoffStationId = getFinalDropoffStationId(selectedAssign);
    const genderFilter = normalizeGenderValue(genderFilterValue);

    const unique = new Set();
    const result = [];

    studentsData.forEach(student => {
        if (!isStudentEligibleForTrip(student)) return;

        const pickupId = Number(student.pickup_station_id);
        const dropoffId = Number(student.dropoff_station_id);

        let matchesRoute = false;

        if (routeFilterValue === "all") {
            matchesRoute =
                stationIds.includes(pickupId) ||
                (finalDropoffStationId !== null && dropoffId === finalDropoffStationId);
        } else if (routeFilterValue === "dropoff") {
            matchesRoute = finalDropoffStationId !== null && dropoffId === finalDropoffStationId;
        } else if (String(routeFilterValue).startsWith("station_")) {
            const stationId = Number(String(routeFilterValue).replace("station_", ""));
            matchesRoute = pickupId === stationId;
        }

        if (!matchesRoute) return;

        if (dayFilterMode === "trip_day_only" && !studentMatchesTripDay(student, tripDate)) {
            return;
        }

        if (genderFilter) {
            const studentGender = normalizeGenderValue(student.gender);
            if (studentGender !== genderFilter) return;
        }

        if (!studentMatchesAbsenceFilter(student, tripDate)) {
            return;
        }

        if (!unique.has(Number(student.id))) {
            unique.add(Number(student.id));
            result.push(student);
        }
    });

    return result;
}

function isElementVisible(el) {
    if (!el) return false;

    const style = window.getComputedStyle(el);
    return (
        style.display !== "none" &&
        style.visibility !== "hidden" &&
        style.opacity !== "0"
    );
}

function getTripStudentIdsArray(trip) {
    if (!trip) return [];

    if (Array.isArray(trip.student_ids)) {
        return trip.student_ids.map(Number).filter(n => !isNaN(n));
    }

    if (typeof trip.student_ids === "string") {
        try {
            const parsed = JSON.parse(trip.student_ids);
            if (Array.isArray(parsed)) {
                return parsed.map(Number).filter(n => !isNaN(n));
            }
        } catch (_) { }

        return trip.student_ids
            .split(",")
            .map(Number)
            .filter(n => !isNaN(n));
    }

    return [];
}

function isSameTripDateAndType(tripA, tripB) {
    return (
        String(tripA?.trip_date || "").trim() === String(tripB?.trip_date || "").trim() &&
        normalizeText(tripA?.trip_type) === normalizeText(tripB?.trip_type)
    );
}

function getConflictTripDisplayName(trip) {
    if (!trip) return "رحلة غير معروفة";

    const resolved = resolveTripData(trip);
    return resolved?.trip_name || buildTripNameFromTrip(trip) || "رحلة غير معروفة";
}

function buildTripConflictMessage(conflicts, candidateTrip) {
    const typeLabel = getTripTypeLabel(candidateTrip.trip_type);
    const parts = [
        `لا يمكن إنشاء الرحلة لأن هناك تعارضًا في نفس التاريخ ونوع الرحلة.`,
        `التاريخ: ${candidateTrip.trip_date}`,
        `نوع الرحلة: ${typeLabel}`
    ];

    if (conflicts.driverConflictTrip) {
        const conflictTripName = getConflictTripDisplayName(conflicts.driverConflictTrip);
        const driverName =
            conflicts.driverName ||
            getDriverById(candidateTrip.driver_id)?.name_driver ||
            "السائق المحدد";

        parts.push(`السائق "${driverName}" مرتبط مسبقًا بالرحلة "${conflictTripName}".`);
    }

    if (conflicts.studentConflictDetails.length) {
        const studentsText = conflicts.studentConflictDetails
            .map(item => `- ${item.student_name} (${item.trip_name})`)
            .join("\n");

        parts.push(`الطلاب التاليون مرتبطون مسبقًا برحلات أخرى في نفس التاريخ ونوع الرحلة:\n${studentsText}`);
    }

    return parts.join("\n\n");
}

function findTripConflicts(candidateTrip, tripsList = []) {
    const selectedStudentIds = Array.isArray(candidateTrip.student_ids)
        ? candidateTrip.student_ids.map(Number).filter(n => !isNaN(n))
        : [];

    const comparableTrips = tripsList.filter(existingTrip => {
        if (!existingTrip) return false;

        const sameDateAndType = isSameTripDateAndType(existingTrip, candidateTrip);
        if (!sameDateAndType) return false;

        if (
            candidateTrip.id &&
            existingTrip.id &&
            Number(existingTrip.id) === Number(candidateTrip.id)
        ) {
            return false;
        }

        return true;
    });

    let driverConflictTrip = null;

    if (candidateTrip.driver_id) {
        driverConflictTrip = comparableTrips.find(existingTrip =>
            Number(existingTrip.driver_id) === Number(candidateTrip.driver_id)
        ) || null;
    }

    const studentConflictMap = new Map();

    comparableTrips.forEach(existingTrip => {
        const existingStudentIds = getTripStudentIdsArray(existingTrip);

        existingStudentIds.forEach(studentId => {
            if (!selectedStudentIds.includes(Number(studentId))) return;
            if (studentConflictMap.has(Number(studentId))) return;

            const student = studentsData.find(s => Number(s.id) === Number(studentId));
            studentConflictMap.set(Number(studentId), {
                student_id: Number(studentId),
                student_name: student?.name || `طالب #${studentId}`,
                trip_id: existingTrip.id ?? null,
                trip_name: getConflictTripDisplayName(existingTrip)
            });
        });
    });

    const studentConflictDetails = Array.from(studentConflictMap.values());

    return {
        hasConflict: Boolean(driverConflictTrip) || studentConflictDetails.length > 0,
        driverConflictTrip,
        driverName: getDriverById(candidateTrip.driver_id)?.name_driver || "",
        studentConflictDetails,
        studentConflictIds: studentConflictDetails.map(item => item.student_id)
    };
}

function validateTripConflictsAgainstTrips(candidateTrip, tripsList, showAlert = true) {
    const conflicts = findTripConflicts(candidateTrip, tripsList);

    if (!conflicts.hasConflict) return true;

    if (showAlert) {
        alert(buildTripConflictMessage(conflicts, candidateTrip));
    }

    return false;
}

async function confirmCreateTrip() {
    const tripData = collectTripFormData();
    if (!tripData) return;

    const candidateTrip = buildCandidateTripForConflictCheck(tripData);

    const latestTrips = await fetchTripsForConflictValidation();
    const latestValidationPassed = validateTripConflictsAgainstTrips(candidateTrip, latestTrips, true);
    if (!latestValidationPassed) {
        await FetchDrive();
        return;
    }

    const nowISO = new Date().toISOString();

    const newTrip = {
        trip_name: tripData.trip_name,
        trip_type: tripData.trip_type,
        trip_date: tripData.trip_date,
        trip_time: tripData.trip_time,
        deadline: tripData.deadline,

        assign_id: Number(tripData.assign_id),
        bus_id: Number(tripData.bus_id),
        driver_id: tripData.driver_id ? Number(tripData.driver_id) : null,

        created_by: currentUserId ? Number(currentUserId) : null,
        created_at: nowISO,
        updated_at: nowISO,

        student_ids: tripSelectedStudentIds.map(id => Number(id))
    };

    try {
        const response = await fetch(TRIPS_API, {
            method: "POST",
            headers: {
                "Content-Type": "application/json",
                "Accept": "application/json"
            },
            body: JSON.stringify(newTrip)
        });

        const savedTrip = await readJsonResponse(response);

        if (!response.ok) {
            throw new Error(
                savedTrip?.message ||
                `فشل حفظ الرحلة. Status: ${response.status}`
            );
        }

        closeTripConfirmModal();
        closeCreateTripModal();
        await FetchDrive();

        alert("تم إنشاء الرحلة بنجاح");
    } catch (error) {
        console.error("Trip Save Error:", error);
        alert(error.message || "حدث خطأ أثناء حفظ الرحلة");
    }
}

function isSameDateOnly(date1, date2) {
    if (!date1 || !date2) return false;
    return formatDateForInput(date1) === formatDateForInput(date2);
}

function addDaysToDate(date, days) {
    const result = new Date(date);
    result.setDate(result.getDate() + days);
    return result;
}

function getTripDateMatch(trip) {
    if (tripsDateFilter === "all") return true;
    if (!trip?.trip_date) return false;

    const today = new Date();
    const todayOnly = new Date(today.getFullYear(), today.getMonth(), today.getDate());

    const tripDate = new Date(`${trip.trip_date}T00:00:00`);
    if (isNaN(tripDate.getTime())) return false;

    const tomorrowOnly = addDaysToDate(todayOnly, 1);
    const weekEnd = addDaysToDate(todayOnly, 6);

    if (tripsDateFilter === "today") {
        return isSameDateOnly(tripDate, todayOnly);
    }

    if (tripsDateFilter === "tomorrow") {
        return isSameDateOnly(tripDate, tomorrowOnly);
    }

    if (tripsDateFilter === "this_week") {
        return tripDate >= todayOnly && tripDate <= weekEnd;
    }

    if (tripsDateFilter === "upcoming") {
        return tripDate >= todayOnly;
    }

    if (tripsDateFilter === "past") {
        return tripDate < todayOnly;
    }

    return true;
}

function getTripCapacityCategory(trip) {
    const studentsCount = Array.isArray(trip.student_ids) ? trip.student_ids.length : 0;
    const capacity = Number(trip.bus_capacity || 0);

    if (!capacity) {
        return studentsCount === 0 ? "empty" : "low";
    }

    const ratio = studentsCount / capacity;

    if (studentsCount === 0) return "empty";
    if (ratio >= 1) return "full";
    if (ratio <= 0.5) return "low";
    return "medium";
}

function getTripCapacityMatch(trip) {
    if (tripsCapacityFilter === "all") return true;
    return getTripCapacityCategory(trip) === tripsCapacityFilter;
}

function fillTripsFilterSelect(selectId, items, selectedValue, defaultLabel) {
    const select = document.getElementById(selectId);
    if (!select) return;

    const uniqueMap = new Map();

    items.forEach(item => {
        const value = String(item.value ?? "").trim();
        const label = String(item.label ?? "").trim();

        if (!value || !label) return;
        if (!uniqueMap.has(value)) {
            uniqueMap.set(value, label);
        }
    });

    select.innerHTML = `<option value="all">${defaultLabel}</option>`;

    Array.from(uniqueMap.entries())
        .sort((a, b) => a[1].localeCompare(b[1], "ar"))
        .forEach(([value, label]) => {
            select.innerHTML += `<option value="${value}">${label}</option>`;
        });

    const hasSelected = Array.from(select.options).some(option => option.value === selectedValue);
    select.value = hasSelected ? selectedValue : "all";
}

        })),

    fillTripsFilterSelect(

            value: trip.created_by || trip.created_by_name,
            label: trip.created_by_name || "غير محدد"

        tripsCreatorFilter,

    tripsDriverFilter = driverSelect ? driverSelect.value : "all";
    tripsRouteFilter = routeSelect ? routeSelect.value : "all";
    tripsCreatorFilter = creatorSelect ? creatorSelect.value : "all";
}

========================= */
function RenderType() {
    const container = getEl("trips-list-container");
    if (!container) return;

    if (!Array.isArray(localTripsData) || !localTripsData.length) {
        populateTripsAdvancedFilterOptions([]);

        container.innerHTML = `
            <div class="trips-empty-state99">
                <i class="bi bi-bus-front"></i>
                <h4>لا توجد رحلات محفوظة حاليًا</h4>
                <p>قم بإنشاء رحلة جديدة وستظهر هنا في جدول منظم.</p>
            </div>
        `;
        return;
    }

    const resolvedTrips = localTripsData
        .map(trip => resolveTripData(trip))
        .filter(Boolean);

    populateTripsAdvancedFilterOptions(resolvedTrips);

    const filteredTrips = resolvedTrips.filter(trip => {
        const matchesType =
            tripsTypeFilter === "all" ||
            normalizeText(trip.trip_type) === normalizeText(tripsTypeFilter);

        const driverValue = String(trip.driver_id || trip.driver_name || "");
        const routeValue = String(trip.route_name || "");
        const creatorValue = String(trip.created_by || trip.created_by_name || "");

        const matchesDriver =
            tripsDriverFilter === "all" ||
            driverValue === String(tripsDriverFilter);

        const matchesRoute =
            tripsRouteFilter === "all" ||
            routeValue === String(tripsRouteFilter);

        const matchesCreator =
            tripsCreatorFilter === "all" ||
            creatorValue === String(tripsCreatorFilter);

        const matchesDate = getTripDateMatch(trip);
        const matchesCapacity = getTripCapacityMatch(trip);

        const searchText = normalizeText(`
            ${trip.trip_name || ""}
            ${trip.route_name || ""}
            ${trip.driver_name || ""}
            ${trip.bus_label || ""}
            ${trip.trip_type_label || ""}
            ${trip.trip_date || ""}
            ${trip.created_by_name || ""}
        `);

        const matchesSearch = !tripsSearchQuery || searchText.includes(tripsSearchQuery);

        return (
            matchesType &&
            matchesDriver &&
            matchesRoute &&
            matchesCreator &&
            matchesDate &&
            matchesCapacity &&
            matchesSearch
        );
    });

    container.innerHTML = filteredTrips.length
        ? `
            <div class="trips-results-count99 mb-2">
                عدد الرحلات المعروضة: <strong>${filteredTrips.length}</strong>
            </div>

            <div class="trips-table-wrap99">
                <table class="trips-table99">
                    <thead>
                        <tr>
                            <th>#</th>
                            <th>اسم الرحلة</th>
                            <th>نوع الرحلة</th>
                            <th>التاريخ</th>
                            <th>الوقت</th>
                            <th>السائق</th>
                            <th>المنشئ</th>
                            <th>عدد الطلاب</th>
                            <th>التفاصيل</th>
                        </tr>
                    </thead>
                    <tbody>
                        ${filteredTrips.map((trip, index) => {
                            const studentsCount = Array.isArray(trip.student_ids) ? trip.student_ids.length : 0;

                            return `
                                <tr>
                                    <td>${index + 1}</td>
                                    <td class="trip-title-cell99">${trip.trip_name || "بدون اسم"}</td>
                                    <td>
                                        <span class="trip-type-chip99 ${trip.trip_type === "pickup" ? "pickup" : "dropoff"}">
                                            ${trip.trip_type_label || "غير محدد"}
                                        </span>
                                    </td>
                                    <td>${trip.trip_date || "غير محدد"}</td>
                                    <td>${formatTime(trip.trip_time || DEFAULT_TRIP_TIME)}</td>
                                    <td>${trip.driver_name || "غير محدد"}</td>
                                    <td>${trip.created_by_name || "غير محدد"}</td>
                                    <td>${studentsCount}</td>
                                    <td>
                                        <button
                                            type="button"
                                            class="trip-table-details-btn99"
                                            onclick="openTripDetailsModal(${trip.id})"
                                        >
                                            <i class="bi bi-eye-fill"></i>
                                        </button>
                                    </td>
                                </tr>
                            `;
                        }).join("")}
                    </tbody>
                </table>
            </div>
        `
        : `
            <div class="trips-empty-state99">
                <i class="bi bi-search"></i>
                <h4>لا توجد نتائج</h4>
                <p>جرّب تغيير البحث أو الفلاتر المستخدمة.</p>
            </div>
        `;
}
